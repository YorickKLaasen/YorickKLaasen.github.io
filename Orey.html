GameConcept

The concept of the game is to get the highest score possible just like in arcade games.
You get a higher score the longer you survive. there is obstacles that you have to avoid since they slow you down.
If you slow down to much a miner will catch you and you will lose.
There is also a Boss that appears at a score of 1000.
You can also choose to unlock achievements which dont give a reward but some people enjoy achievement hunting.

MainMenu.cs


using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.SceneManagement;
using UnityEngine.UI;
using TMPro;
using Unity.VisualScripting;

public class MainMenu : MonoBehaviour
{
    // Titlescreen music made by Eiten Epstein: https://www.youtube.com/watch?v=MjyzJ_MaIH0&t=37s
    // Math music https://www.youtube.com/watch?v=ubNfkpbxXUs&t=8s / https://www.youtube.com/watch?v=FtJ03-g83Yg
    // Main music and boss by HeatleyBros https://www.youtube.com/watch?v=meLQ3Nn5VPM

    public Animator transition;
    public AudioClip startSound;
    public AudioClip insertCoin;
    public AudioClip denied;
    public AudioClip achievementsSound;
    public AudioSource audioSource;

    public bool buttonPressed = true;
    int playerCoins = 3;
    int score = 0;
    int timesPlayed = 0;
    const string playerCoinsKey = "playerCoins";

    [SerializeField] TextMeshProUGUI coinText;
    [SerializeField] TextMeshProUGUI warningText;

    private void Start()
    {
        if (PlayerPrefs.HasKey(playerCoinsKey))
        {
            playerCoins = PlayerPrefs.GetInt(playerCoinsKey, 0);
        }
        else
        {
            PlayerPrefs.SetInt(playerCoinsKey, playerCoins);
        }
        timesPlayed = PlayerPrefs.GetInt("timesPlayed", timesPlayed);
        coinText.text = "" + playerCoins.ToString();
        PlayerPrefs.Save();
    }
    public void OnQuitClick()
    {
        Application.Quit();
    }
    public void OnStartClick()
    {
        if (playerCoins >= 1)
        {
            if (buttonPressed == true)
            {
                timesPlayed++;
                score = 0;
                PlayerPrefs.SetInt("Score", score);
                PlayerPrefs.SetInt("timesPlayed", timesPlayed);
                PlayerPrefs.Save();
                buttonPressed = false;
                CoinRemove();
                StartCoroutine(PlaySoundAndLoadGame());
            }
        }
        else
        {
            audioSource.PlayOneShot(denied);
            warningText.text = "You dont have enough coins to play. Go challenge yourself at the Coin Machine to earn some coins!";
        }
    }
    public void OnAchievementClick()
    {
        StartCoroutine(AchievementLoad());
    }
    IEnumerator AchievementLoad()
    {
        audioSource.PlayOneShot(achievementsSound);
        transition.SetTrigger("Start");
        yield return new WaitForSeconds(1.5f);
        SceneManager.LoadScene("AchievementsMenu");
    }
    public void OnInfoClick()
    {
        StartCoroutine(InfoLoad());
    }
    IEnumerator InfoLoad()
    {
        audioSource.PlayOneShot(achievementsSound);
        transition.SetTrigger("Start");
        yield return new WaitForSeconds(1.5f);
        SceneManager.LoadScene("InformationMenu");
    }
    IEnumerator PlaySoundAndLoadGame()
    {
        audioSource.PlayOneShot(startSound);
        yield return new WaitForSeconds(2.5f);
        SceneManager.LoadScene("Game");
    }
    public void CoinRemove()
    {
        playerCoins--;
        PlayerPrefs.SetInt("playerCoins", playerCoins);
        PlayerPrefs.Save();
        coinText.text = playerCoins.ToString();
    }
    public void OnInsertCoinClick()
    {
        StartCoroutine(LoadInsertCoin());
    }
    IEnumerator LoadInsertCoin()
    {
        audioSource.PlayOneShot(insertCoin);
        transition.SetTrigger("Start");
        yield return new WaitForSeconds(1f);
        SceneManager.LoadScene("CoinMachine");
    }
    private void OnApplicationQuit()
    {
        playerCoins = 3;
        PlayerPrefs.SetInt(playerCoinsKey, playerCoins);
        PlayerPrefs.Save();
    }
}

        PlayerPrefs.SetInt(coinsGainedKey, coinsGained);
        PlayerPrefs.SetInt(correctAnswersKey, answerAchievement);
        PlayerPrefs.SetInt(timesPlayedKey, timesPlayed);

    private void AchievementCheck()
    {
        highScore1Text.color = highScore >= 5000 ? Color.green : Color.red;
    }

    public void OnBackClick() => SceneManager.LoadScene("MainMenu");

CoinGainMachine.cs


                        public class CoinGainMachine : MonoBehaviour
{
    public Animator transition, easterEgg;
    [SerializeField] TMP_InputField playerInput;
    [SerializeField] TextMeshProUGUI question, answerStatus, correctAnswersText, timerText, coinObtainedText;
    [SerializeField] Button confirmAnswerButton, restartButton;
    int number1, number2, randomNum, questionAnswer, playerAnswer, correctAnswers, answerAchievement, coinsGained, viscachaSecret;
    const string playerCoinsKey = "playerCoins";
    float quizTimer;
    string numberOperator;

    public void OnBackClick() => StartCoroutine(LoadBack());
    IEnumerator LoadBack()
    {
        transition.SetTrigger("Start");
        yield return new WaitForSeconds(1f);
        SceneManager.LoadScene("MainMenu");
    }

    private void Start()
    {
        QuestionShow();
        CalculateQuestion();
        StartCoroutine(Timer());
        coinsGained = PlayerPrefs.GetInt("coinsGained", coinsGained);
        answerAchievement = PlayerPrefs.GetInt("correctAnswers", answerAchievement);
        viscachaSecret = PlayerPrefs.GetInt("viscachaSecret", viscachaSecret);
    }

    void Update()
    {
        if (quizTimer > 0) quizTimer -= Time.deltaTime;
        timerText.text = quizTimer > 0 ? "Time Left: " + (int)quizTimer : "";
    }

    public void onRestartClick()
    {
        confirmAnswerButton.interactable = true;
        answerStatus.text = "";
        correctAnswers = 0;
        correctAnswersText.text = "Correct Answers: " + correctAnswers;
        StartCoroutine(Timer());
        QuestionShow();
    }

    public void OnConfirmAnswer()
    {
        if (int.TryParse(playerInput.text, out playerAnswer))
        {
            if (playerAnswer == questionAnswer)
            {
                answerStatus.text = "Correct!";
                QuestionShow();
                CalculateQuestion();
                playerInput.text = "";
                correctAnswers++;
                answerAchievement++;
                PlayerPrefs.SetInt("correctAnswers", answerAchievement);
                PlayerPrefs.Save();
                correctAnswersText.text = "Correct Answers: " + correctAnswers;
                if (correctAnswers % 3 == 0)
                {
                    coinObtainedText.text = "You Gained a Coin Yippie!";
                    CoinGaining();
                }
            }
            else if (playerAnswer == 229191381)
            {
                viscachaSecret++;
                PlayerPrefs.SetInt("viscachaSecret", viscachaSecret);
                PlayerPrefs.Save();
                answerStatus.text = "oh?";
                easterEgg.SetTrigger("Start");
            }
            else
            {
                answerStatus.text = "Incorrect! Try Again";
                playerInput.text = "";
            }
        }
        else answerStatus.text = "Only Enter Numbers Please";
    }

    public void CoinGaining()
    {
        int playerCoins = PlayerPrefs.GetInt(playerCoinsKey, 0);
        PlayerPrefs.SetInt(playerCoinsKey, ++playerCoins);
        PlayerPrefs.SetInt("coinsGained", ++coinsGained);
        PlayerPrefs.Save();
    }

    public void QuestionShow()
    {
        number1 = Random.Range(100, 1000);
        number2 = Random.Range(0, 1000);
        numberOperator = Random.Range(0, 2) == 0 ? "+" : "-";
        CalculateQuestion();
        question.text = $"{number1}{numberOperator}{number2}";
    }

    public int CalculateQuestion()
    {
        questionAnswer = numberOperator == "+" ? number1 + number2 : number1 - number2;
        return questionAnswer;
    }

    IEnumerator Timer()
    {
        quizTimer = 50;
        while (quizTimer > 0)
        {
            timerText.text = "Time Left: " + quizTimer.ToString();
            yield return new WaitForSeconds(1);
            quizTimer--;
        }
        answerStatus.text = "Time is UP!";
        confirmAnswerButton.interactable = false;
    }
}

GameOverUI.cs


                        public class GameOverUI : MonoBehaviour
{
    [SerializeField] TMP_Text highScoreText, scoreText, coinText, warningText;
    const string playerCoinsKey = "playerCoins", highScoreKey = "highscore", scoreKey = "Score";
    int highScore, playerCoins, score;

    private void Start()
    {
        highScore = PlayerPrefs.GetInt(highScoreKey, 0);
        playerCoins = PlayerPrefs.GetInt(playerCoinsKey, 0);
        score = PlayerPrefs.GetInt(scoreKey, 0);
        PlayerPrefs.SetInt(playerCoinsKey, playerCoins);
        PlayerPrefs.SetInt(highScoreKey, highScore);
        PlayerPrefs.SetInt(scoreKey, score);
        PlayerPrefs.Save();
        scoreText.text = $"Score: {score}";
        highScoreText.text = $"HighScore: {highScore}";
        coinText.text = playerCoins.ToString();
    }

    public void OnRetryClick()
    {
        if (playerCoins >= 1)
        {
            playerCoins--;
            PlayerPrefs.SetInt(playerCoinsKey, playerCoins);
            PlayerPrefs.SetInt(scoreKey, 0);
            PlayerPrefs.Save();
            coinText.text = $"Coins: {playerCoins}";
            SceneManager.LoadScene("Game");
        }
        else warningText.text = "You don't have enough coins. Go to the Coin Machine for more.";
    }

    public void OnQuitClick() => SceneManager.LoadScene("MainMenu");
}
